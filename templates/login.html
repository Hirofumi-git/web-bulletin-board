from flask import Flask, render_template, request, redirect, url_for, session from datetime import datetime app = Flask(__name__) app.secret_key = 'your_secret_key' # 仮ユーザー辞書 users = { 'user1': 'pass1', 'user2': 'pass2' } # 投稿データ保持用（メモリ上） posts = [] # ログインページ @app.route('/login', methods=['GET', 'POST']) def login(): if request.method == 'POST': username = request.form['username'] password = request.form['password'] if username in users and users[username] == password: session['username'] = username return redirect(url_for('index')) else: return "ログイン失敗" return render_template('login.html') # ログアウト @app.route('/logout') def logout(): session.pop('username', None) return redirect(url_for('login')) # 掲示板ページ @app.route('/', methods=['GET', 'POST']) def index(): if 'username' not in session: return redirect(url_for('login')) if request.method == 'POST': content = request.form['content'] if content: posts.append({ 'user': session['username'], 'content': content, 'time': datetime.now().strftime('%Y-%m-%d %H:%M:%S') }) return render_template('index.html', posts=posts) if __name__ == '__main__': app.run(host='0.0.0.0', port=8080, debug=True)
